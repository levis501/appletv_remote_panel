HITBOX_COLORMAP.md



    Red - (255, 0, 0) - power button
    Green - (0, 255, 0) - dpad center
    Blue - (0, 0, 255) - dpad up
    Yellow - (255, 255, 0) - dpad left
    Cyan - (0, 255, 255) - dpad right
    Magenta - (255, 0, 255) - dpad down
    Orange - (255, 165, 0) - connected light
    Purple - (128, 0, 128) - back button
    Brown - (165, 42, 42) - home button
    Pink - (255, 192, 203) - play/pause button
    Gray - (128, 128, 128) - volume up
    Black - (0, 0, 0) - volume down
    White - (255, 255, 255) - open app selector
